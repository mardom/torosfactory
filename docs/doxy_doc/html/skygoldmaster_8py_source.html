87 ##
88 # Return background mean and std. dev. of sky background.
89 #
90 # Calculate the background (sky) mean value and a measure of its standard deviation.
91 # Background is considered anything below 'noiseLvl' sigmas.
92 # goodPixMask is a mask containing the good pixels that should be considered.
93 # Return mean, std_dev
94 #
95 def bkgNoiseSigma(dataImg, noiseLvl = 3.0, goodPixMask = None):
96  m = dataImg[goodPixMask].mean()
97  s = dataImg[goodPixMask].std()
98 
99  prevSgm = 2*s #This will make the first while condition true
100  tol = 1E-2
101  while abs(prevSgm - s)/s > tol:
102  prevSgm = s
103  bkgMask = np.logical_and(dataImg < m + noiseLvl*s, dataImg > m - noiseLvl*s)
104  if goodPixMask is not None: bkgMask = np.logical_and(bkgMask, goodPixMask)
105  m = dataImg[bkgMask].mean()
106  s = dataImg[bkgMask].std()
107 
108  return m, s

271 # Find a rotation + translation transformation between pairs of points using a RANSAC algorithm.
272 def findAffineTransform(src_pts, dst_pts):
273  assert src_pts.shape == dst_pts.shape
274  data = np.hstack((src_pts, dst_pts)) #a set of observed data points
275  transfModel = LinearTransformModel() #a model that can be fitted to data points
276 
277  minNDataPoints = 2 #the minimum number of data values required to fit the model
278  maxPixDist = 5. #a threshold value for determining when a data point fits a model
279  minNInliers = 10 #the number of close data values required to assert that a model fits well to data
280 
281  #w is the probability of choosing an inlier each time a single point is selected
282  #w = number of inliers in data / number of points in data
283  w = 45./len(src_pts)
284 
285  #p is the probability that the RANSAC algorithm in some iteration selects only inliers
286  #from the input data set when it chooses the n points from which the model parameters are estimated
287  p = 0.99
288 
289  #the maximum number of iterations allowed in the algorithm
290  maxNIterations = 2 * int(math.log(1. - p) / math.log(1. - w ** minNDataPoints))
291  if maxNIterations > 1000: maxNIterations = 1000
292 
293  mask = np.zeros(len(src_pts), dtype = 'bool')
294  try:
295  M, maskDict = ransac.ransac(data,
296  transfModel,
297  minNDataPoints,
298  maxNIterations * 2,
299  maxPixDist,
300  minNInliers,
301  debug=False,
302  return_all=True)
303  except:
304  M = None
305 
306  else:
307  mask[maskDict['inliers']] = True
308  #improve M with all inliers
309  M = transfModel.fit(data[mask])
310 
311  return M, mask

314 class LinearTransformModel:
315  def __init__(self): pass
316  def fit(self, data):
317  A = []
318  y = []
319  for pt in data:
320  x1 = pt[0]
321  x2 = pt[1]
322  A.append([x1, x2, 1, 0])
323  A.append([x2, -x1, 0, 1])
324  y.append(pt[2])
325  y.append(pt[3])
326 
327  A = np.matrix(A)
328  y = np.matrix(y)
329 
330  sol, resid, rank, sv = np.linalg.lstsq(A,y.T)
331  c = sol.item(0)
332  s = sol.item(1)
333  t1 = sol.item(2)
334  t2 = sol.item(3)
335  approxM = np.array([[c, s, t1],[-s, c, t2]])
336  return approxM
