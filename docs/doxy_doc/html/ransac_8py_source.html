38 # fit model parameters to data using the RANSAC algorithm
39 #
40 # This implementation written from pseudocode found at
41 # http://en.wikipedia.org/w/index.php?title=RANSAC&oldid=116358182
42 #
43 # {{{
44 # Given:
45 # data - a set of observed data points
46 # model - a model that can be fitted to data points
47 # n - the minimum number of data values required to fit the model
48 # k - the maximum number of iterations allowed in the algorithm
49 # t - a threshold value for determining when a data point fits a model
50 # d - the number of close data values required to assert that a model fits well to data
51 # Return:
52 # bestfit - model parameters which best fit the data (or nil if no good model is found)
53 # iterations = 0
54 # bestfit = nil
55 # besterr = something really large
56 # while iterations < k {
57 # maybeinliers = n randomly selected values from data
58 # maybemodel = model parameters fitted to maybeinliers
59 # alsoinliers = empty set
60 # for every point in data not in maybeinliers {
61 # if point fits maybemodel with an error smaller than t
62 # add point to alsoinliers
63 # }
64 # if the number of elements in alsoinliers is > d {
65 # % this implies that we may have found a good model
66 # % now test how good it is
67 # bettermodel = model parameters fitted to all points in maybeinliers and alsoinliers
68 # thiserr = a measure of how well model fits these points
69 # if thiserr < besterr {
70 # bestfit = bettermodel
71 # besterr = thiserr
72 # }
73 # }
74 # increment iterations
75 # }
76 # return bestfit
77 # }}}
78 #
79 def ransac(data,model,n,k,t,d,debug=False,return_all=False):
80  iterations = 0
81  bestfit = None
82  #besterr = numpy.inf
83  best_inlier_idxs = None
84  while iterations < k:
85  maybe_idxs, test_idxs = random_partition(n,data.shape[0])
86  maybeinliers = data[maybe_idxs,:]
87  test_points = data[test_idxs,:]
88  maybemodel = model.fit(maybeinliers)
89  test_err = model.get_error( test_points, maybemodel)
90  also_idxs = test_idxs[test_err < t] # select indices of rows with accepted points
91  alsoinliers = data[also_idxs,:]
92  if len(alsoinliers) > d:
93  betterdata = numpy.concatenate( (maybeinliers, alsoinliers) )
94  bestfit = model.fit(betterdata)
95  #better_errs = model.get_error( betterdata, bettermodel)
96  #thiserr = numpy.mean( better_errs )
97  #if thiserr < besterr:
98  #bestfit = bettermodel
99  #besterr = thiserr
100  best_inlier_idxs = numpy.concatenate( (maybe_idxs, also_idxs) )
101  break
102  iterations+=1
103  if bestfit is None:
104  raise ValueError("did not meet fit acceptance criteria")
105  if return_all:
106  return bestfit, {'inliers':best_inlier_idxs}
107  else:
108  return bestfit

112 def random_partition(n,n_data):
113  all_idxs = numpy.arange( n_data )
114  numpy.random.shuffle(all_idxs)
115  idxs1 = all_idxs[:n]
116  idxs2 = all_idxs[n:]
117  return idxs1, idxs2

120 # linear system solved using linear least squares
121 #
122 # This class serves as an example that fulfills the model interface
123 # needed by the ransac() function.
124 #
125 #
126 class LinearLeastSquaresModel:
127  def __init__(self,input_columns,output_columns,debug=False):
128  self.input_columns = input_columns
129  self.output_columns = output_columns
130  self.debug = debug
131  def fit(self, data):
132  A = numpy.vstack([data[:,i] for i in self.input_columns]).T
133  B = numpy.vstack([data[:,i] for i in self.output_columns]).T
134  x,resids,rank,s = scipy.linalg.lstsq(A,B)
135  return x
136  def get_error( self, data, model):
137  A = numpy.vstack([data[:,i] for i in self.input_columns]).T
138  B = numpy.vstack([data[:,i] for i in self.output_columns]).T
139  B_fit = scipy.dot(A,model)
140  err_per_point = numpy.sum((B-B_fit)**2,axis=1) # sum squared error per row
141  return err_per_point
