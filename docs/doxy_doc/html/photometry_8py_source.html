57 # Get the aperture flux at the (RA, Dec) position for the radecList array
58 def getFluxesForRADec(radecList, image, goodMask, the_wcs, bkg_level = 0, apRadius = 3.5):
59  pix_ctr = []
60  good_pix_mask = []
61  for aPoint in radecList:
62  try:
63  pixPoint = the_wcs.all_world2pix([aPoint],0,tolerance=1E-3)[0]
64  x,y = pixPoint.astype('int')
65  h, w = image.shape
66  if(x > w or x < 0 or y < 0 or y > h):
67  good_pix_mask.append(False)
68  elif(goodMask[y,x] == False):
69  good_pix_mask.append(False)
70  else:
71  pix_ctr.append(pixPoint)
72  good_pix_mask.append(True)
73  except:
74  good_pix_mask.append(False)
75 
76  pix_ctr = np.array(pix_ctr)
77  good_pix_mask = np.array(good_pix_mask)
78 
79  apertures = CircularAperture(pix_ctr, apRadius)
80  fluxes = aperture_photometry(image, apertures)
81  fluxes -= bkg_level * np.pi * apRadius**2
82 
83  return fluxes, good_pix_mask

87 # Return the best fit to the y-intercept of the x,y data assuming a is fixed, ignoring outliers.
88 #
89 # This is a least square problem to solve the equation y=ax+b when a is fixed (not fitted)
90 # to some value. It will also clip out recursively data points that are away from nsig std dev
91 # from the current iteration.
92 # It stops when it doesn't find any more outliers away from nsig std. deviations.
93 # Return b
94 #
95 def sigmaClipIntercepFit(x, y, nsig = 3, a = -2.5):
96  b = -sum((a*x - y))/len(x)
97  sig = math.sqrt(sum((x*a + b - y)**2)/len(x))
98  gpts_mask = np.abs(x*a + b - y) < nsig*sig
99 
100  gx = x.copy()
101  gy = y.copy()
102  MAX_ITER = 1000
103  i = 0
104  while(sum(gpts_mask) != len(gx) and i < MAX_ITER):
105  gx = gx[gpts_mask]
106  gy = gy[gpts_mask]
107  b = -sum((a*gx - gy))/len(gx)
108  sig = math.sqrt(sum((gx*a + b - gy)**2)/len(gx))
109  gpts_mask = np.abs(gx*a + b - gy) < nsig*sig
110  i += 1
111  return b

114 ##
115 # Return background mean and std-dev of sky background.
116 #
117 # Calculate the background (sky) mean value and a measure of its standard deviation.
118 # Background is considered anything below 'noiseLvl' sigmas.
119 # goodPixMask is a mask containing the good pixels that should be considered.
120 # Return mean, std_dev
121 #
122 def bkgNoiseSigma(dataImg, noiseLvl = 3.0, goodPixMask = None):
123  m = dataImg[goodPixMask].mean()
124  s = dataImg[goodPixMask].std()
125 
126  prevSgm = 2*s #This will make the first while condition true
127  tol = 1E-2
128  while abs(prevSgm - s)/s > tol:
129  prevSgm = s
130  bkgMask = np.logical_and(dataImg < m + noiseLvl*s, dataImg > m - noiseLvl*s)
131  if goodPixMask != None: bkgMask = np.logical_and(bkgMask, goodPixMask)
132  m = dataImg[bkgMask].mean()
133  s = dataImg[bkgMask].std()
134 
135  return (m, s)
